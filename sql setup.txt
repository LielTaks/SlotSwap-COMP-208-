CREATE TABLE timetable (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    day VARCHAR(20),
    time VARCHAR(20),
    activity VARCHAR(100),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
